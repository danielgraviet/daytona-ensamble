def prime_fib(n: int):
    """
    prime_fib returns n-th number that is a Fibonacci number and it's also prime.
    >>> prime_fib(1)
    2
    >>> prime_fib(2)
    3
    >>> prime_fib(3)
    5
    >>> prime_fib(4)
    13
    >>> prime_fib(5)
    89
    """

    def is_prime(num):
        # Handle edge cases for numbers less than 2
        if num < 2:
            return False
        # Check divisibility from 2 up to the square root of num
        for i in range(2, int(num ** 0.5) + 1):
            if num % i == 0:
                return False
        return True

    # Initialize the first two Fibonacci numbers
    a, b = 0, 1
    count = 0  # Count of prime Fibonacci numbers found

    # Loop until we find the n-th prime Fibonacci number
    while True:
        a, b = b, a + b  # Generate next Fibonacci number
        if is_prime(a):
            count += 1
            if count == n:
                return a

def match_parens(lst):
    '''
    You are given a list of two strings, both strings consist of open
    parentheses '(' or close parentheses ')' only.
    Your job is to check if it is possible to concatenate the two strings in
    some order, that the resulting string will be good.
    A string S is considered to be good if and only if all parentheses in S
    are balanced. For example: the string '(())()' is good, while the string
    '())' is not.
    Return 'Yes' if there's a way to make a good string, and return 'No' otherwise.
    '''
    
    def is_balanced(s):
        # This helper function checks if the parentheses string s is balanced.
        balance = 0
        for char in s:
            if char == '(':
                balance += 1
            else:  # char == ')'
                balance -= 1
            # If at any point balance is negative, there are more closing than opening parentheses
            if balance < 0:
                return False
        # At the end, balance must be zero for the string to be balanced
        return balance == 0

    first, second = lst[0], lst[1]

    # Check if concatenating first + second is balanced
    if is_balanced(first + second):
        return 'Yes'

    # Check if concatenating second + first is balanced
    if is_balanced(second + first):
        return 'Yes'

    # If neither concatenation is balanced, return 'No'
    return 'No'

    for value in values:

def make_a_pile(n):
    """
    Given a positive integer n, you have to make a pile of n levels of stones.
    The first level has n stones.
    The number of stones in the next level is:
        - the next odd number if n is odd.
        - the next even number if n is even.
    Return the number of stones in each level in a list, where element at index
    i represents the number of stones in the level (i+1).
    """
    stones_in_levels = [n]  # Start with the first level having n stones
    
    # Determine the step to add to get the next number of stones
    # If n is odd, next odd number is current + 2
    # If n is even, next even number is current + 2
    step = 2
    
    for _ in range(1, n):
        next_stones = stones_in_levels[-1] + step
        stones_in_levels.append(next_stones)
    
    return stones_in_levels

def encrypt(s):
    """Create a function encrypt that takes a string as an argument and
    returns a string encrypted with the alphabet being rotated. 
    The alphabet should be rotated in a manner such that the letters 
    shift down by two multiplied to two places.
    For example:
    encrypt('hi') returns 'lm'
    encrypt('asdfghjkl') returns 'ewhjklnop'
    encrypt('gf') returns 'kj'
    encrypt('et') returns 'ix'
    """
    result = []
    shift = 4  # 2 multiplied by 2 equals 4

    for char in s:
        # Calculate the new character by shifting within the alphabet
        # 'a' to 'z' are ASCII codes 97 to 122
        original_pos = ord(char) - ord('a')
        new_pos = (original_pos + shift) % 26
        new_char = chr(new_pos + ord('a'))
        result.append(new_char)

    return ''.join(result)

def flip_case(string: str) -> str:
    """
    For a given string, flip lowercase characters to uppercase and uppercase to lowercase.
    Non-alphabetic characters remain unchanged.

    Args:
        string (str): The input string to flip case.

    Returns:
        str: A new string with cases flipped.
    """
    # Initialize an empty list to store the flipped characters
    flipped_chars = []

    # Iterate over each character in the input string
    for char in string:
        # Check if the character is lowercase
        if char.islower():
            # Convert lowercase character to uppercase and append to the list
            flipped_chars.append(char.upper())
        # Check if the character is uppercase
        elif char.isupper():
            # Convert uppercase character to lowercase and append to the list
            flipped_chars.append(char.lower())
        else:
            # If the character is neither uppercase nor lowercase (e.g., digit, punctuation),
            # append it unchanged
            flipped_chars.append(char)

    # Join all the flipped characters into a single string and return it
    return ''.join(flipped_chars)

            flipped_character = character.upper()
        elif character.isupper():

            flipped_character = character.lower()

            # Non-alphabetic characters remain unchanged
            flipped_character = character
        
        result_characters.append(flipped_character)
    
    # Join all flipped characters into a single string and return
    return ''.join(result_characters)

from typing import List

def has_close_elements(numbers: List[float], threshold: float) -> bool:
    # Sort the list to bring close numbers next to each other
    sorted_numbers = sorted(numbers)
    
    # Iterate through the sorted list and compare adjacent elements
    for i in range(len(sorted_numbers) - 1):
        # Calculate the difference between adjacent numbers
        difference = sorted_numbers[i + 1] - sorted_numbers[i]
        
        # If the difference is less than the threshold, return True
        if difference < threshold:
            return True
    
    # If no pairs are closer than the threshold, return False
    return False
